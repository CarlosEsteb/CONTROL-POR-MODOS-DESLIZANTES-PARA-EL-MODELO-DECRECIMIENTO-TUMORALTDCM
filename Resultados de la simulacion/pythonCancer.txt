#RK4
def RungeKutta4(f,x0,t0,tf,dt):
	t=np.arange(t0,tf,dt)
	nt=t.size
	nx=x0.size
	x=np.zeros((nx,nt))
	x[:,0]=x0

	for k in range(nt-1):
		k1=dt*f(t[k],x[:,k])
		k2=dt*f(t[k]+dt/2,x[:,k]+k1/2)
		k3=dt*f(t[k]+dt/2,x[:,k]+k2/2)
		k4=dt*f(t[k]+dt,x[:,k]+k3)
		dx=(k1+2*k2+2*k3+k4)/6
		x[:,k+1]=x[:,k]+dx

	return x,t
